workflow buildBamIndexTest {
	call buildBamIndex
}

task buildBamIndex {
	File inputBam
	Int memoryGb
	Int diskSpaceGb
	String fileId = basename(inputBam, ".bam")

	command <<<
		java -jar /usr/gitc/picard.jar BuildBamIndex \
		I=${inputBam} \
		O="${fileId}.bai"
	>>>

	output {
		File bamIndex="${fileId}.bai"
	}

	runtime {
		docker: "broadinstitute/genomes-in-the-cloud:2.3.1-1500064817"
		memory: "${memoryGb} GB"
		cpu: "1"
		disks: "local-disk ${diskSpaceGb} HDD"
	}
}
